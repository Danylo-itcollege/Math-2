### 2.2 –î–æ–ø–æ–º—ñ–∂–Ω—ñ —Ñ—É–Ω–∫—Ü—ñ—ó
```python
def print_header(text):
    display(Markdown(f"**{text}**"))

def plot_feasible_region(A, b, x_opt=None):
    """–í—ñ–∑—É–∞–ª—ñ–∑–∞—Ü—ñ—è –æ–±–ª–∞—Å—Ç—ñ –¥–æ–ø—É—Å—Ç–∏–º–∏—Ö —Ä—ñ—à–µ–Ω—å"""
    x = np.linspace(0, 50, 100)
    plt.figure(figsize=(10,6))

    for i in range(A.shape[0]):
        y = (b[i] - A[i,0]*x)/A[i,1]
        plt.plot(x, y, label=f'{A[i,0]:.1f}x1 + {A[i,1]:.1f}x2 ‚â§ {b[i]:.1f}')
    
    if x_opt is not None:
        plt.scatter(x_opt[0], x_opt[1], color='red', s=100, 
                   label=f'–û–ø—Ç–∏–º—É–º ({x_opt[0]:.1f}, {x_opt[1]:.1f})')

    plt.xlabel('x1 (–ü—Ä–æ–¥—É–∫—Ç A)')
    plt.ylabel('x2 (–ü—Ä–æ–¥—É–∫—Ç B)')
    plt.legend()
    plt.grid()
    plt.show()
```

### 2.3 –ó–∞–¥–∞—á–∞ 1: –ú–∞–∫—Å–∏–º—ñ–∑–∞—Ü—ñ—è –ø—Ä–∏–±—É—Ç–∫—É
```python
print_header("–ó–∞–¥–∞—á–∞ 1: –ú–∞–∫—Å–∏–º—ñ–∑–∞—Ü—ñ—è –ø—Ä–∏–±—É—Ç–∫—É")

c = [-60, -40]  # –ú—ñ–Ω—É—Å, –±–æ linprog –≤–∏–∫–æ–Ω—É—î –º—ñ–Ω—ñ–º—ñ–∑–∞—Ü—ñ—é
A = [[3, 2], [1, 3], [2, 2]]
b = [180, 120, 150]

res = linprog(c, A_ub=A, b_ub=b, bounds=[(0, None), (0, None)], method='highs')

print(f"–°—Ç–∞—Ç—É—Å: {'–£—Å–ø—ñ—à–Ω–æ' if res.success else '–ü–æ–º–∏–ª–∫–∞'}")
print(f"–û–ø—Ç–∏–º–∞–ª—å–Ω–∏–π –ø–ª–∞–Ω: x1 = {res.x[0]:.2f}, x2 = {res.x[1]:.2f}")
print(f"–ú–∞–∫—Å–∏–º–∞–ª—å–Ω–∏–π –ø—Ä–∏–±—É—Ç–æ–∫: {-res.fun:.2f} –≥—Ä–Ω")
plot_feasible_region(np.array(A), np.array(b), res.x)
```
